145 public static double Incomplete(double a, double b, double x)
146 {
147 double aa, bb, t, xx, xc, w, y;
148 bool flag;
149
150 // validate inputs
151 if (a <= 0.0)
152 {
153 throw new ArgumentOutOfRangeException("a", "Lower limit a must be greater than zero.");
154 }
155
156 if (b <= 0.0)
157 {
158 throw new ArgumentOutOfRangeException("b", "Upper limit b must be greater than zero.");
159 }
160
161 if (x <= 0.0 || x >= 1.0)
162 {
163 if (x == 0.0)
164 {
165 return 0.0;
166 }
167
168 if (x == 1.0)
169 {
170 return 1.0;
171 }
172
173 throw new ArgumentOutOfRangeException("x", "The x value must be between 0 and 1.");
174 }
175
176 flag = false;
177 if (b * x <= 1.0 && x <= 0.95)
178 {
179 t = PowerSeries(a, b, x);
180 return t;
181 }
182
183 w = 1.0 - x;
184 if (x > a / (a + b))
185 {
186 flag = true;
187 aa = b;
188 bb = a;
189 xc = x;
190 xx = w;
191 }
192 else
193 {
194 aa = a;
195 bb = b;
196 xc = w;
197 xx = x;
198 }
199
200 if (flag && bb * xx <= 1.0 && xx <= 0.95)
201 {
202 t = PowerSeries(aa, bb, xx);
203 if (t <= Tools.DoubleMachineEpsilon)
204 {
205 t = 1.0 - Tools.DoubleMachineEpsilon;
206 }
207 else
208 {
209 t = 1.0 - t;
210 }
211
212 return t;
213 }
214
215 y = xx * (aa + bb - 2.0) - (aa - 1.0);
216 if (y < 0.0)
217 {
218 w = Incbcf(aa, bb, xx);
219 }
220 else
221 {
222 w = Incbd(aa, bb, xx) / xc;
223 }
224
225 y = aa * Math.Log(xx);
226 t = bb * Math.Log(xc);
227 if (aa + bb < GammaMax && Math.Abs(y) < LogMax && Math.Abs(t) < LogMax)
228 {
229 t = Math.Pow(xc, bb);
230 t *= Math.Pow(xx, aa);
231 t /= aa;
232 t *= w;
233 t *= Gamma.Function(aa + bb) / (Gamma.Function(aa) * Gamma.Function(bb));
234 if (flag)
235 {
236 if (t <= Tools.DoubleMachineEpsilon)
237 {
238 t = 1.0 - Tools.DoubleMachineEpsilon;
239 }
240 else
241 {
242 t = 1.0 - t;
243 }
244 }
245
246 return t;
247 }
248
249 y += t + Gamma.LogGamma(aa + bb) - Gamma.LogGamma(aa) - Gamma.LogGamma(bb);
250 y += Math.Log(w / aa);
251 if (y < LogMin)
252 {
253 t = 0.0;
254 }
255 else
256 {
257 t = Math.Exp(y);
258 }
259
260 if (flag)
261 {
262 if (t <= Tools.DoubleMachineEpsilon)
263 {
264 t = 1.0 - Tools.DoubleMachineEpsilon;
265 }
266 else
267 {
268 t = 1.0 - t;
269 }
270 }
271
272 return t;
273 }

496 public static double IncompleteInverse(double aa, double bb, double yy0)
497 {
498 const double LogMin = -745.13321910194122d;
499 double a, b, y0, d, y, x, x0, x1, lgm, yp, di, dithresh, yl, yh;
500 int i, dir;
501 bool nflg;
502 bool rflg;
503
504 if (yy0 <= 0)
505 return 0.0;
506 if (yy0 >= 1.0)
507 return 1.0;
508
509 if (aa <= 1.0 || bb <= 1.0)
510 {
511 nflg = true;
512 dithresh = 4.0 * Tools.DoubleMachineEpsilon;
513 rflg = false;
514 a = aa;
515 b = bb;
516 y0 = yy0;
517 x = a / (a + b);
518 y = Incomplete(a, b, x);
519 goto ihalve;
520 }
521 else
522 {
523 nflg = false;
524 dithresh = 0.0001;
525 }
526
527 yp = -Normal.StandardZ(yy0);
528 if (yy0 > 0.5d)
529 {
530 rflg = true;
531 a = bb;
532 b = aa;
533 y0 = 1.0 - yy0;
534 yp = -yp;
535 }
536 else
537 {
538 rflg = false;
539 a = aa;
540 b = bb;
541 y0 = yy0;
542 }
543
544 lgm = (yp * yp - 3.0) / 6.0;
545 x0 = 2.0 / (1.0 / (2.0 * a - 1.0) + 1.0 / (2.0 * b - 1.0));
546 y = yp * Math.Sqrt(x0 + lgm) / x0 - (1.0 / (2.0 * b - 1.0) - 1.0 / (2.0 * a - 1.0)) * (lgm + 5.0 / 6.0 - 2.0 / (3.0 * x0));
547 y = 2.0 * y;
548 if (y < LogMin)
549 {
550 x0 = 1.0;
551 throw new ArithmeticException("Underflow");
552 }
553
554 x = a / (a + b * Math.Exp(y));
555 y = Incomplete(a, b, x);
556 yp = (y - y0) / y0;
557 if (Math.Abs(yp) < 0.01d)
558 {
559 goto newt;
560 }
561
562 ihalve:
563
564 x0 = 0.0;
565 yl = 0.0;
566 x1 = 1.0;
567 yh = 1.0;
568 di = 0.5;
569 dir = 0;
570 for (i = 0; i < 400; i++)
571 {
572 if (i != 0)
573 {
574 x = x0 + di * (x1 - x0);
575 if (x == 1.0)
576 x = 1.0 - Tools.DoubleMachineEpsilon;
577 y = Incomplete(a, b, x);
578 yp = (x1 - x0) / (x1 + x0);
579 if (Math.Abs(yp) < dithresh)
580 {
581 x0 = x;
582 goto newt;
583 }
584 }
585
586 if (y < y0)
587 {
588 x0 = x;
589 yl = y;
590 if (dir < 0)
591 {
592 dir = 0;
593 di = 0.5;
594 }
595 else if (dir > 1)
596 {
597 di = 0.5 * di + 0.5;
598 }
599 else
600 {
601 di = (y0 - y) / (yh - yl);
602 }
603
604 dir += 1;
605 if (x0 > 0.75)
606 {
607 if (rflg)
608 {
609 rflg = false;
610 a = aa;
611 b = bb;
612 y0 = yy0;
613 }
614 else
615 {
616 rflg = true;
617 a = bb;
618 b = aa;
619 y0 = 1.0 - yy0;
620 }
621
622 x = 1.0 - x;
623 y = Incomplete(a, b, x);
624 goto ihalve;
625 }
626 }
627 else
628 {
629 x1 = x;
630 if (rflg && x1 < Tools.DoubleMachineEpsilon)
631 {
632 x0 = 0.0;
633 goto done;
634 }
635
636 yh = y;
637 if (dir > 0)
638 {
639 dir = 0;
640 di = 0.5;
641 }
642 else if (dir < -1)
643 {
644 di = 0.5 * di;
645 }
646 else
647 {
648 di = (y - y0) / (yh - yl);
649 }
650
651 dir -= 1;
652 }
653 }
654
655 if (x0 >= 1.0)
656 {
657 x0 = 1.0 - Tools.DoubleMachineEpsilon;
658 goto done;
659 }
660
661 if (x == 0.0)
662 throw new ArithmeticException("Underflow");
663
664 newt:
665 if (nflg)
666 {
667 goto done;
668 }
669
670 x0 = x;
671 lgm = Gamma.LogGamma(a + b) - Gamma.LogGamma(a) - Gamma.LogGamma(b);
672 for (i = 0; i < 10; i++)
673 {
674 if (i != 0)
675 y = Incomplete(a, b, x0);
676
677 d = (a - 1.0) * Math.Log(x0) + (b - 1.0) * Math.Log(1.0 - x0) + lgm;
678 if (d < LogMin)
679 throw new ArithmeticException("Underflow");
680 d = Math.Exp(d);
681 d = (y - y0) / d;
682 x = x0;
683 x0 -= d;
684 if (x0 <= 0.0)
685 throw new ArithmeticException("underflow");
686 if (x0 >= 1.0)
687 {
688 x0 = 1.0 - Tools.DoubleMachineEpsilon;
689 goto done;
690 }
691
692 if (Math.Abs(d / x0) < 64.0 * Tools.DoubleMachineEpsilon)
693 {
694 goto done;
695 }
696 }
697
698 done:
699 if (rflg)
700 {
701 if (x0 <= double.Epsilon)
702 {
703 x0 = 1.0 - double.Epsilon;
704 }
705 else
706 {
707 x0 = 1.0 - x0;
708 }
709 }
710
711 return x0;
712 }

735 public static double PowerSeries(double a, double b, double x)
736 {
737 double s, t, u, v, n, t1, z, ai;
738 ai = 1.0 / a;
739 u = (1.0 - b) * x;
740 v = u / (a + 1.0);
741 t1 = v;
742 t = u;
743 n = 2.0;
744 s = 0.0;
745 z = Tools.DoubleMachineEpsilon * ai;
746 while (Math.Abs(v) > z)
747 {
748 u = (n - b) * x / n;
749 t *= u;
750 v = t / (a + n);
751 s += v;
752 n += 1.0;
753 }
754
755 s += t1;
756 s += ai;
757 u = a * Math.Log(x);
758 if (a + b < GammaMax && Math.Abs(u) < LogMax)
759 {
760 t = Gamma.Function(a + b) / (Gamma.Function(a) * Gamma.Function(b));
761 s = s * t * Math.Pow(x, a);
762 }
763 else
764 {
765 t = Gamma.LogGamma(a + b) - Gamma.LogGamma(a) - Gamma.LogGamma(b) + u + Math.Log(s);
766 if (t < LogMin)
767 {
768 s = 0.0;
769 }
770 else
771 {
772 s = Math.Exp(t);
773 }
774 }
775
776 return s;

798 public static double IncompleteRatio(double x, double p, double q, double beta)
799 {
800
801 // Define accuracy and initialize parameters
802 double acu = 0.000000000000001d;
803 bool indx;
804 double psq;
805 double cx;
806 double xx;
807 double pp;
808 double qq;
809 double term;
810 double ai;
811 int ns;
812 double rx;
813 double temp;
814 double b;
815
816 // Check the input arguments.
817 if (p <= 0.0 | q <= 0.0)
818 {
819 throw new Exception("Invalid inputs! P and Q must be positive.");
820 }
821
822 if (x < 0.0 | x > 1.0)
823 {
824 throw new Exception("Invalid inputs! X must be between 0 and 1.");
825 }
826
827 // Special cases.
828 if (x == 0.0 | x == 1.0)
829 {
830 return x;
831 }
832
833 // Change tail if necessary and determine S.
834 psq = p + q;
835 cx = 1.0 - x;
836 if (p < psq * x)
837 {
838 xx = cx;
839 cx = x;
840 pp = q;
841 qq = p;
842 indx = true;
843 }
844 else
845 {
846 xx = x;
847 pp = p;
848 qq = q;
849 indx = false;
850 }
851
852 term = 1.0;
853 ai = 1.0;
854 b = 1.0;
855 ns = (int)(qq + cx * psq);
856
857 // Use the Soper reduction formula.
858 rx = xx / cx;
859 temp = qq - ai;
860 if (ns == 0) rx = xx;
861 rx = xx / cx;
862 three:
863 ;
864 temp = qq - ai;
865 if (ns == 0) rx = xx;
866 four:
867 ;
868 term = term * temp * rx / (pp + ai);
869 b += term;
870 temp = Math.Abs(term);
871 if (temp <= acu & temp <= acu * b)
872 goto five;
873 ai += 1;
874 ns -= 1;
875 if (ns >= 0)
876 goto three;
877 temp = psq;
878 psq += 1d;
879 goto four;
880
881 // calculate result
882 five:
883 ;
884 b = b * Math.Exp(pp * Math.Log(xx) + (qq - 1.0) * Math.Log(cx) - beta) / pp;
885 if (indx)
886 b = 1.0 - b;
887 return b;
888 }
