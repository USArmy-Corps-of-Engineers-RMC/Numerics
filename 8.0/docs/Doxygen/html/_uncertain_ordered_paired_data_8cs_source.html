296 public UncertainOrderedPairedData(XElement el)
297 {
298 // Get Order
299 if (el.Attribute("X_Strict") != null)
300 bool.TryParse(el.Attribute("X_Strict").Value, out _strictX);
301 if (el.Attribute("Y_Strict") != null)
302 bool.TryParse(el.Attribute("Y_Strict").Value, out _strictY);
303 // Get Strictness
304 if (el.Attribute("X_Order") != null)
305 Enum.TryParse(el.Attribute("X_Order").Value, out _orderX);
306 if (el.Attribute("Y_Order") != null)
307 Enum.TryParse(el.Attribute("Y_Order").Value, out _orderY);
308 // Distribution type
309 Distribution = UnivariateDistributionType.Deterministic;
310 if (el.Attribute("Distribution") != null)
311 {
312 var argresult = Distribution;
313 Enum.TryParse(el.Attribute("Distribution").Value, out argresult);
314 Distribution = argresult;
315 }
316 // new prop
317
318 if (el.Attribute(nameof(AllowDifferentDistributionTypes)) != null)
319 {
320 bool.TryParse(el.Attribute(nameof(AllowDifferentDistributionTypes)).Value, out _allowDifferentDistributionTypes);
321 // Get Ordinates
322 var curveEl = el.Element("Ordinates");
323 _uncertainOrdinates = new List<UncertainOrdinate>();
324 foreach (XElement ord in curveEl.Elements(nameof(UncertainOrdinate)))
325 _uncertainOrdinates.Add(new UncertainOrdinate(ord));
326 }
327 else
328 {
329 var curveEl = el.Element("Ordinates");
330 var xData = new List<double>();
331 var yData = new List<UnivariateDistributionBase>();
332 if (curveEl != null)
333 {
334 foreach (XElement o in curveEl.Elements("Ordinate"))
335 {
336 double.TryParse(o.Attribute("X").Value, out var xout);
337 xData.Add(xout);
338 var dist = UnivariateDistributionFactory.CreateDistribution(Distribution);
339 var props = dist.GetParameterPropertyNames;
340 var paramVals = new double[(props.Count())];
341 for (int i = 0; i < props.Count(); i++)
342 {
343 double.TryParse(o.Attribute(props[i]).Value, out var result);
344 paramVals[i] = result;
345 }
346
347 dist.SetParameters(paramVals);
348 yData.Add(dist);
349 }
350 }
351 //
352 _uncertainOrdinates = new List<UncertainOrdinate>();
353 for (int i = 0; i < xData.Count; i++)
354 _uncertainOrdinates.Add(new UncertainOrdinate(xData[i], yData[i].Clone()));
355 }
356
357 Validate();
358 }

434 private bool OrdinateValid(int index, bool lookBackward = true)
435 {
436 if (index < 0 || index > _uncertainOrdinates.Count - 1)
437 return false;
438 // Look backward
439 if (lookBackward == true & index > 0)
440 {
441 if (_uncertainOrdinates[index].OrdinateValid(_uncertainOrdinates[index - 1], StrictX, StrictY, OrderX, OrderY, false, AllowDifferentDistributionTypes) == false)
442 return false;
443 }
444 // Look forward
445 if (index < _uncertainOrdinates.Count - 1)
446 {
447 if (_uncertainOrdinates[index].OrdinateValid(_uncertainOrdinates[index + 1], StrictX, StrictY, OrderX, OrderY, true, AllowDifferentDistributionTypes) == false)
448 return false;
449 }
450 // Passed the test
451 return true;
452 }

476 public static bool operator ==(UncertainOrderedPairedData left, UncertainOrderedPairedData right)
477 {
478 // Check for null arguments. Keep in mind null == null
479 if (left is null && right is null)
480 {
481 return true;
482 }
483 else if (left is null)
484 {
485 return false;
486 }
487 else if (right is null)
488 {
489 return false;
490 }
491 if ((left._uncertainOrdinates == null) && (right._uncertainOrdinates == null))
492 {
493 return true;
494 }
495 else if (left._uncertainOrdinates == null)
496 {
497 return false;
498 }
499 else if (right._uncertainOrdinates == null)
500 {
501 return false;
502 }
503 if (left.Count != right.Count)
504 return false;
505 for (int i = 0, loopTo = left.Count - 1; i <= loopTo; i++)
506 {
507 if (left._uncertainOrdinates[i].X != right._uncertainOrdinates[i].X)
508 return false;
509 if (left._uncertainOrdinates[i].Y == right._uncertainOrdinates[i].Y == false)
510 return false;
511 }
512 return true;

621 public void RemoveRange(int[] rowIndicesToRemove)
622 {
623 if ((rowIndicesToRemove == null) || rowIndicesToRemove.Count() == 0)
624 return;
625 // Need to remove from last to first. Can't assume that the array is sorted. need to clone array to ensure the original index array is maintained.
626 int[] sortedIndices = (int[])rowIndicesToRemove.Clone();
627 Array.Sort(sortedIndices);
628 var removedItems = new List<UncertainOrdinate>();
629 bool isContinuous = true;
630 for (int i = sortedIndices.Count() - 1; i >= 0; i -= 1)
631 {
632 if (i < sortedIndices.Count() - 1 && sortedIndices[i] != sortedIndices[i + 1] - 1)
633 isContinuous = false;
634 removedItems.Add(_uncertainOrdinates[sortedIndices[i]]);
635 _uncertainOrdinates.RemoveAt(sortedIndices[i]);
636 }
637 Validate();
638 if (SupressCollectionChanged == false)
639 {
640 if (isContinuous)
641 {
642 CollectionChanged?.Invoke(this, new NotifyCollectionChangedEventArgs(NotifyCollectionChangedAction.Remove, removedItems, sortedIndices[0]));
643 }
644 else
645 {
646 CollectionChanged?.Invoke(this, new NotifyCollectionChangedEventArgs(NotifyCollectionChangedAction.Remove, removedItems));
647 }
648 }
