96 protected override void Optimize()
97 {
98 // Define variables
99 bool cancel = false;
100 // Bracket problem
101 double ax = LowerBound, bx = 0.5 * (UpperBound + LowerBound), cx = UpperBound;
102 double R = 0.61803399, C = 1.0 - R;
103 double x1, x2;
104 double x0 = ax;
105 double x3 = cx;
106 if (Math.Abs(cx - bx) > Math.Abs(bx - ax))
107 {
108 x1 = bx;
109 x2 = bx + C * (cx - bx);
110 }
111 else
112 {
113 x2 = bx;
114 x1 = bx - C * (bx - ax);
115 }
116 double f1 = Evaluate(new double[] { x1 }, ref cancel);
117 double f2 = Evaluate(new double[] { x2 }, ref cancel);
118 while (Math.Abs(x3 - x0) > AbsoluteTolerance || Math.Abs(x3 - x0) > RelativeTolerance * (Math.Abs(x1) + Math.Abs(x2)))
119 {
120 if (f2 < f1)
121 {
122 x0 = x1;
123 x1 = x2;
124 x2 = R * x2 + C * x3;
125 f1 = f2;
126 f2 = Evaluate(new double[] { x2 }, ref cancel);
127 }
128 else
129 {
130 x3 = x2;
131 x2 = x1;
132 x1 = R * x1 + C * x0;
133 f2 = f1;
134 f1 = Evaluate(new double[] { x1 }, ref cancel);
135 }
136 if (cancel == true) return;
137 Iterations += 1;
138 if (Iterations >= MaxIterations)
139 {
140 UpdateStatus(OptimizationStatus.MaximumIterationsReached);
141 return;
142 }
143 }
144 UpdateStatus(OptimizationStatus.Success);
145 }
146
147 }
