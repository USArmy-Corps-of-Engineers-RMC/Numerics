340 public void SolveParameters()
341 {
342 if (_parametersValid == false) return;
343 if (_parametersSolved == true) return;
344
345 double fifth = Percentile5th;
346 double fiftieth = Percentile50th;
347 double ninetyFifth = Percentile95th;
348
349 if (fifth == fiftieth && fiftieth == ninetyFifth)
350 {
351 _beta = GeneralizedBeta.PERT(fifth, fiftieth, ninetyFifth);
352 _parametersSolved = true;
353 return;
354 }
355
356 _beta = GeneralizedBeta.PERT(fifth, fiftieth, ninetyFifth);
357 // Set wide bounds
358 double min = fifth - (ninetyFifth - fifth) * 2;
359 double max = ninetyFifth + (ninetyFifth - fifth) * 2;
360 var Initials = new double[] { _beta.Min, _beta.Mode, _beta.Max };
361 var Lowers = new double[] { min, min, min };
362 var Uppers = new double[] { max, max, max };
363
364 // Objective function is the sum of squared errors
365 double sse(double[] x)
366 {
367 var dist = new Pert();
368 try
369 {
370 dist = new Pert(x[0], x[1], x[2]);
371 if (dist.ParametersValid == false) return double.MaxValue;
372
373 double SSE = 0d;
374 SSE += Math.Pow(fifth - dist.InverseCDF(0.05), 2d);
375 SSE += Math.Pow(fiftieth - dist.InverseCDF(0.5), 2d);
376 SSE += Math.Pow(ninetyFifth - dist.InverseCDF(0.95), 2d);
377 return SSE;
378 }
379 catch
380 {
381 return double.MaxValue;
382 }
383
384 }
385
386 var solver = new BFGS(sse, 3, Initials, Lowers, Uppers);
387 solver.RelativeTolerance = 1E-4;
388 solver.AbsoluteTolerance = 1E-8;
389 solver.ReportFailure = false;
390 solver.Minimize();
391 var solution = solver.BestParameterSet.Values;
392 _beta = GeneralizedBeta.PERT(solution[0], solution[1], solution[2]);
393 _parametersSolved = true;
394 }

427 public override double InverseCDF(double probability)
428 {
429 // Validate parameters
430 if (_parametersValid == false) ValidateParameters(Percentile5th, Percentile50th, Percentile95th, true);
431 if (_parametersSolved == false) SolveParameters();
432 //
433 var x = _beta.Min;
434 if (x < MinAllowableValue) x = MinAllowableValue;
435 if (x > MaxAllowableValue) x = MaxAllowableValue;
436 // These checks are done specifically for an application where a
437 // user inputs min = max = mode
438 if (_beta.Min == _beta.Max) return x;
439 if (double.IsNaN(_beta.Mode)) return x;
440 //
441 try
442 {
443 x = _beta.InverseCDF(probability);
444 }
445 catch (ArithmeticException)
446 {
447 return x;
448 }
449 //
450 if (x < MinAllowableValue) x = MinAllowableValue;
451 if (x > MaxAllowableValue) x = MaxAllowableValue;
452 return x;
