76 public LUDecomposition(Matrix A)
77 {
78 // Given a square matrix A[0..n-1][0..n-1], this routine replaces it by the LU decomposition of a
79 // row-wise permutation of itself. Index[0..n-1] is an output vector that records the row permutation effected by the
80 // partial pivoting; d is output as +-1 depending on whether the number of row interchanges was even or odd, respectively.
81 // This routine is used in combination with Solve to solve linear equations or inverse a matrix.
82
83 this.A = new Matrix(A.ToArray());
84 LU = new Matrix(A.ToArray());
85 n = A.NumberOfRows;
86 index = new int[n];
87 var vv = new double[n];
88 double TINY = 1.0E-40d;
89 int imax, i, j, k;
90 double big, temp;
91 d = 1.0d;
92
93 // Loop over rows to get the implicit scaling information.
94 for (i = 0; i < n; i++)
95 {
96 big = 0.0d;
97 for (j = 0; j < n; j++)
98 {
99 temp = Math.Abs(LU[i, j]);
100 if (temp > big) big = temp;
101 }
102
103 if (big == 0.0d) throw new ArgumentException("Singular matrix in LU decomposition.");
104 vv[i] = 1.0d / big;
105 }
106 // This is the outermost k-i-j loop.
107 for (k = 0; k < n; k++)
108 {
109 big = 0.0d;
110 imax = k;
111 for (i = k; i < n; i++)
112 {
113 temp = vv[i] * Math.Abs(LU[i, k]);
114 // Is the figure of merit for the pivot better than the best so far?
115 if (temp > big)
116 {
117 big = temp;
118 imax = i;
119 }
120 }
121
122 if (k != imax)
123 {
124 // Do we need to interchange rows? Yes, do so...
125 for (j = 0; j < n; j++)
126 {
127 temp = LU[imax, j];
128 LU[imax, j] = LU[k, j];
129 LU[k, j] = temp;
130 }
131 d = -d;
132 vv[imax] = vv[k];
133 }
134
135 index[k] = imax;
136
137 // If the pivot element is zer0, the matrix is singular (at least to the precision of the algorithm.)
138 // For some applications on singular matrices, it is desirable to substitute TINY for zero.
139 if (LU[k, k] == 0.0d) LU[k, k] = TINY;
140
141 //Perform Gaussian Elimination step
142 for (i = k + 1; i < n; i++)
143 {
144 LU[i, k] /= LU[k, k]; // True Lower triangular matrix
145 for (j = k + 1; j < n; j++)
146 LU[i, j] -= LU[i, k] * LU[k, j]; // True Upper triangular matrix
147 }
148 }
149 }

169 public Vector Solve(Vector B)
170 {
171 // Solves the set of n linear equations A*x=b using the stored LU decomposition of A.
172 // b[0..n-1] is input as the right-hand side vector b, while x returns the solution vector x;
173 // This routine takes into account the possibility that b will begin with many zero elements,
174 // so it is efficient for use in matrix inversion.
175 int i, ii = 0, ip, j;
176 double sum;
177 var x = new Vector(n);
178 if (B.Length != n)
179 {
180 throw new ArgumentOutOfRangeException(nameof(B), "The vector b must have the same number of rows as the matrix A.");
181 }
182 for (i = 0; i < n; i++) x[i] = B[i];
183 for (i = 0; i < n; i++)
184 {
185 ip = index[i];
186 sum = x[ip];
187 x[ip] = x[i];
188 if (ii != 0)
189 for (j = ii - 1; j < i; j++) sum -= LU[i,j] * x[j];
190 else if (sum != 0.0)
191 ii = i + 1;
192 x[i] = sum;
193 }
194 for (i = n - 1; i >= 0; i--)
195 {
196 sum = x[i];
197 for (j = i + 1; j < n; j++) sum -= LU[i,j] * x[j];
198 x[i] = sum / LU[i,i];
199 }
200 return x;
201 }

207 public Matrix Solve(Matrix B)
208 {
209 int i, j, m = B.NumberOfColumns;
210 var x = new Matrix(n);
211 var xx = new Vector(n);
212 if (B.NumberOfRows != n)
213 {
214 throw new ArgumentOutOfRangeException(nameof(B), "The vector b must have the same number of rows as the matrix A.");
215 }
216 for (j = 0; j < m; j++)
217 {
218 for (i = 0; i < n; i++) { xx[i] = B[i, j]; }
219 xx = Solve(xx);
220 for (i = 0; i < n; i++) { x[i, j] = xx[i]; }
221 }
222 return x;
223 }

228 public Matrix InverseA()
229 {
230 int i, j;
231 var Ainv = new Matrix(n);
232 for (i = 0; i < n; i++)
233 {
234 for (j = 0; j < n; j++) Ainv[i, j] = 0d;
235 Ainv[i, i] = 1d;
236 }
237 return Solve(Ainv);
